계정의 기본 네임을 입력받고 생성하고 싶은 개수를 받아
계정을 생성하는 스크립트를 작성하세요
for 문을 이용



-------------------------------------------
답)
-------------------------------------------
#!/bin/sh
echo "계정의 이름을 입력하세요"
read name
echo "만들고 싶은 계정의 숫자를 입력하세요"
read num
for (( i=1 ; i <=$num ; i++))
do
        useradd $name$i
        echo "$name$i 계정을 생성했습니다!"
done
exit 0
