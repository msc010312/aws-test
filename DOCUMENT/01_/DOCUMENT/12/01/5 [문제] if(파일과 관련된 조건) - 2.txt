#!/bin/sh
echo "생성할  파일을 입력하세요."
read fname
touch $fname
echo "부여할 권한을 입력하세요(r|rw)"
read perm
if [ $perm = "rw" ] || [ $perm = "r" ];
then
        chmod a=$perm $fname

else
        echo "잘못입력하셨습니다. 종료 합니다"
fi
exit 0
