특정 경로를 입력 후 그 경로가 디렉토리인지 파일인지 
구분하는 스크립트를 작성하세요


-------------------------------------------------
답
-------------------------------------------------
#!/bin/sh
echo "경로를 입력하세요"
read path
if [ -d $path ]
        then
        echo "디렉토리 입니다"
        else
        echo "파일입니다"
fi
exit 0
